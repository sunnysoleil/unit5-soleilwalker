<script>
        let searchBox = document.querySelector('#searcher');
        console.log(searchBox.addEventListener("input", null).value);
        let search = searchBox.value;
        let gallery = document.getElementById('imagesContainer');
        let aList = gallery.getElementsByTagName('a');
    

        function searchBoxer() {
            let a;
            for (let i = 0; i < aList.length; i++) {
                a = aList[i].getAttribute('data-caption');
                if (a.toUpperCase().indexOf(search) > -1 ) {
                    aList[i].style.display = "";
                } else {
                    aList[i].style.display = "none";
                }
            }
        }

        searchBox.addEventListener('input', () => {
            searchBoxer();
        })
